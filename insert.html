âš™ Backend Code (PHP)

<?php // route: user/add

use Classes\Request;
use Classes\Response;
use Tables\User;

$username = Request::post("username");
$password = Request::post("password");
$fullname = Request::post("fullname");

if (! $username) {
    Response::code(404)->message("Username is required")->send();
}

if (! $password) {
    Response::code(404)->message("Password is required")->send();
}

if (! $fullname) {
    Response::code(404)->message("Fullname is required")->send();
}

$result = User::insert([
    "username" => $username,
    "password" => $password,
    "fullname" => $fullname,
]);

Response::code(200)->message("User added")->send();
